    Authentication

    Login.

    Returns access token and authenticated user email, name and default password status. If user.is_default_password is true, User must change password to access routes that require authentication.

    Example request:

    
$client = new \GuzzleHttp\Client();
$response = $client->post(
    'http://localhost/api/login',
    [
        'headers' => [
            'Content-Type' => 'application/json',
            'Accept' => 'application/json',
            'Authorization' => 'Bearer {token}',
        ],
        'json' => [
            'email' => 'jdoe@example.com',
            'password' => 'passwords',
        ],
    ]
);
$body = $response->getBody();
print_r(json_decode((string) $body));
    curl -X POST \
    "http://localhost/api/login" \
    -H "Content-Type: application/json" \
    -H "Accept: application/json" \
    -H "Authorization: Bearer {token}" \
    -d '{"email":"jdoe@example.com","password":"passwords"}'

    const url = new URL(
    "http://localhost/api/login"
);

let headers = {
    "Content-Type": "application/json",
    "Accept": "application/json",
    "Authorization": "Bearer {token}",
};

let body = {
    "email": "jdoe@example.com",
    "password": "passwords"
}

    method: "POST",

    .then(response => response.json())
    .then(json => console.log(json));
